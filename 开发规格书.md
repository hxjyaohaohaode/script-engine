# 互动影游剧本智能体集群 — 开发规格书

> 重要说明：本文档保留了早期方案与生成式脚手架示例，部分路径、接口和模型名已过时。
> 当前可运行实现以 `script-engine/README.md`、`script-engine/backend/api/*.py`、`script-engine/render.yaml` 为准。
> 阅读本文时，请将其中的 `backend/agents/*`、旧接口路径和旧模型名视为历史设计参考，而非当前事实来源。
>
> 本文档是面向AI编程助手（Trae/Cursor等）的完整开发规格书，可直接用于代码实现。
> 目标：构建一个Web应用，支持几万字到150万字的互动影游剧本协作生产。
> 部署：GitHub → Render

---

## 一、技术栈

```
前端:     React 18 + TypeScript + Tailwind CSS + Ant Design
后端:     Python 3.11 + FastAPI + Celery（异步任务）
数据库:   PostgreSQL 15（Render自带）
缓存/消息: Redis（Render自带）
图数据:   用PostgreSQL JSONB + 邻接表替代Neo4j（简化部署）
AI调用:   httpx调用DeepSeek API + MiMo API
部署:     Render（Web Service + PostgreSQL + Redis）
```

**为什么不用Neo4j**：Render不自带Neo4j，外接增加复杂度。伏笔图谱用PostgreSQL的JSONB字段+邻接表就能实现，性能足够。

## 三、数据库设计

### 3.1 PostgreSQL建表语句

```sql
-- 项目表
CREATE TABLE projects (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(200) NOT NULL,
    description TEXT,
    genre VARCHAR(50),                    -- 题材: 权谋/悬疑/科幻/民国...
    style VARCHAR(50),                    -- 风格: 暗黑/轻松/写实...
    target_word_count INTEGER,            -- 目标字数
    player_count VARCHAR(20) DEFAULT 'single',  -- single/multi
    work_mode VARCHAR(20) DEFAULT 'standard',   -- light/standard/heavy
    core_contradiction TEXT,              -- 核心矛盾
    core_truth TEXT,                      -- 核心真相（玩家最终发现的）
    status VARCHAR(20) DEFAULT 'draft',   -- draft/setting/outlining/writing/reviewing/completed
    current_phase INTEGER DEFAULT 0,      -- 当前Phase 0-6
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 角色表
CREATE TABLE characters (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
    char_code VARCHAR(20) NOT NULL,       -- CHAR-001
    name VARCHAR(100) NOT NULL,
    role_type VARCHAR(20),                -- protagonist/antagonist/supporting/minor
    background TEXT,                      -- 背景故事
    core_goal TEXT,                       -- 核心动机
    core_fear TEXT,                       -- 核心恐惧
    surface_image TEXT,                   -- 表面形象
    true_self TEXT,                       -- 真实面目
    language_style TEXT,                  -- 语言风格
    catchphrase TEXT,                     -- 口头禅
    arc_description TEXT,                 -- 角色弧描述
    behavior_inevitable JSONB DEFAULT '[]',   -- 必然行为
    behavior_never JSONB DEFAULT '[]',        -- 绝对不会行为
    behavior_conditional JSONB DEFAULT '[]',  -- 需要铺垫才能行为
    status VARCHAR(20) DEFAULT 'active',
    created_at TIMESTAMP DEFAULT NOW()
);

-- 角色关系表
CREATE TABLE character_relations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
    char_a_id UUID REFERENCES characters(id) ON DELETE CASCADE,
    char_b_id UUID REFERENCES characters(id) ON DELETE CASCADE,
    relation_type VARCHAR(50),            -- structural_type: 核心冲突/表面盟友/表面敌对/枢纽
    trust INTEGER DEFAULT 50,             -- 信任度 0-100
    favor INTEGER DEFAULT 50,             -- 好感度 0-100
    info_known_a_about_b JSONB DEFAULT '[]',  -- A知道B的秘密
    info_known_b_about_a JSONB DEFAULT '[]',  -- B知道A的秘密
    updated_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(project_id, char_a_id, char_b_id)
);

-- 伏笔表
CREATE TABLE foreshadows (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
    fs_code VARCHAR(20) NOT NULL,         -- F-001
    name VARCHAR(200) NOT NULL,
    fs_type VARCHAR(20) NOT NULL,         -- global/chapter/scene/interactive
    surface_layer TEXT,                   -- 表面层含义
    deep_layer TEXT,                      -- 深层含义
    truth_layer TEXT,                     -- 真相层含义
    plant_scene_id UUID,                  -- 埋设场景
    reinforce_scenes JSONB DEFAULT '[]',  -- 强化场景列表
    reveal_scene_id UUID,                 -- 揭露场景
    wow_factor TEXT,                      -- 哇塞效果描述
    player_reaction TEXT,                 -- 预期玩家反应
    depends_on JSONB DEFAULT '[]',        -- 依赖的其他伏笔ID
    enables JSONB DEFAULT '[]',           -- 本伏笔揭露后让哪些伏笔重要
    current_status VARCHAR(20) DEFAULT 'design',  -- design/plant/reinforce/reveal/verify
    reinforce_count INTEGER DEFAULT 0,
    health VARCHAR(20) DEFAULT 'normal',  -- normal/warning/critical
    wow_plans JSONB DEFAULT '[]',         -- 哇塞方案列表
    wow_selected VARCHAR(50),             -- 选中的哇塞方案ID
    created_at TIMESTAMP DEFAULT NOW()
);

-- 伏笔关联表（邻接表替代Neo4j）
CREATE TABLE foreshadow_relations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
    from_fs_id UUID REFERENCES foreshadows(id) ON DELETE CASCADE,
    to_fs_id UUID REFERENCES foreshadows(id) ON DELETE CASCADE,
    relation_type VARCHAR(20) NOT NULL,   -- depends_on/supports/enables/conflicts
    created_at TIMESTAMP DEFAULT NOW()
);

-- 章节表
CREATE TABLE chapters (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
    chapter_number INTEGER NOT NULL,
    title VARCHAR(200),
    summary TEXT,                         -- 一句话核心事件
    outline TEXT,                         -- 详细大纲
    emotion_target INTEGER DEFAULT 5,     -- 本章情感目标
    key_turning_points JSONB DEFAULT '[]', -- 关键转折点
    foreshadow_tasks JSONB DEFAULT '[]',   -- 本章伏笔任务
    branch_structure TEXT,                 -- 分支结构描述
    anchor_scenes JSONB DEFAULT '[]',      -- 锚点场景ID列表 [开头/高潮/结尾]
    status VARCHAR(20) DEFAULT 'draft',
    created_at TIMESTAMP DEFAULT NOW()
);

-- 场景表
CREATE TABLE scenes (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
    chapter_id UUID REFERENCES chapters(id) ON DELETE CASCADE,
    scene_code VARCHAR(30) NOT NULL,      -- CH03-S07-A
    scene_type VARCHAR(30),               -- dialogue/transition/action/choice/convergence
    location VARCHAR(200),
    weather VARCHAR(100),
    time_start VARCHAR(50),               -- 剧本内时间
    time_end VARCHAR(50),
    emotion_level INTEGER DEFAULT 5,      -- 情感强度 0-10
    narration TEXT,                       -- 场景描述
    dialogue JSONB DEFAULT '[]',          -- [{char_id, text, subtext}]
    actions JSONB DEFAULT '[]',           -- 动作指示
    foreshadow_ops JSONB DEFAULT '[]',    -- [{fs_id, op, content}]
    choices JSONB DEFAULT '[]',           -- [{id, text, consequence, next_scene_code, hidden, prerequisites}]
    causal_chain JSONB,                   -- {preconditions, catalyst, direct_result, indirect_result, far_result}
    is_wow_moment BOOLEAN DEFAULT FALSE,
    wow_type VARCHAR(30),
    wow_spec TEXT,
    characters_involved JSONB DEFAULT '[]',
    status VARCHAR(20) DEFAULT 'draft',   -- draft/auditing/rejected/approved/final
    version INTEGER DEFAULT 1,
    audit_reports JSONB DEFAULT '[]',
    human_reviewed BOOLEAN DEFAULT FALSE,
    human_feedback TEXT,
    git_commit_hash VARCHAR(40),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 场景历史版本表
CREATE TABLE scene_versions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    scene_id UUID REFERENCES scenes(id) ON DELETE CASCADE,
    version INTEGER NOT NULL,
    content JSONB NOT NULL,               -- 完整场景内容快照
    audit_report JSONB,
    change_reason VARCHAR(200),
    created_at TIMESTAMP DEFAULT NOW()
);

-- 元素注册表
CREATE TABLE elements (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
    element_type VARCHAR(20) NOT NULL,    -- character/item/location/organization/event/info
    element_code VARCHAR(20) NOT NULL,    -- CHAR-001 / ITEM-001 / LOC-001
    name VARCHAR(200) NOT NULL,
    status VARCHAR(20) DEFAULT 'active',  -- active/dead/destroyed/resolved
    first_appear_scene_code VARCHAR(30),
    last_update_scene_code VARCHAR(30),
    description TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- 信息点表
CREATE TABLE info_points (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
    info_code VARCHAR(20) NOT NULL,       -- I-001
    content TEXT NOT NULL,
    importance VARCHAR(10),               -- low/medium/high/critical
    known_by JSONB DEFAULT '[]',          -- 知道此信息的角色ID列表
    planted_in_scene_code VARCHAR(30),
    created_at TIMESTAMP DEFAULT NOW()
);

-- 审计记录表
CREATE TABLE audit_records (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
    scene_id UUID REFERENCES scenes(id) ON DELETE CASCADE,
    audit_type VARCHAR(20) NOT NULL,      -- scene/full
    checker_results JSONB NOT NULL,       -- 程序化检测器结果
    llm_results JSONB,                    -- LLM审计结果
    creative_scores JSONB,                -- 创意评分
    overall_result VARCHAR(10),           -- pass/fail
    issues JSONB DEFAULT '[]',
    suggestions JSONB DEFAULT '[]',
    created_at TIMESTAMP DEFAULT NOW()
);

-- Agent任务队列表
CREATE TABLE agent_tasks (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
    task_type VARCHAR(30) NOT NULL,       -- write_scene/audit_scene/design_foreshadow/full_audit
    assigned_to VARCHAR(20) NOT NULL,     -- agent名称
    status VARCHAR(20) DEFAULT 'pending', -- pending/running/completed/failed
    priority INTEGER DEFAULT 5,
    payload JSONB NOT NULL,               -- 任务参数
    result JSONB,                         -- 任务结果
    error_message TEXT,
    retry_count INTEGER DEFAULT 0,
    max_retries INTEGER DEFAULT 3,
    created_at TIMESTAMP DEFAULT NOW(),
    started_at TIMESTAMP,
    completed_at TIMESTAMP
);

-- 情感曲线表
CREATE TABLE emotion_curve (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
    chapter_id UUID REFERENCES chapters(id),
    scene_id UUID REFERENCES scenes(id),
    target_emotion INTEGER,               -- 目标情感强度
    actual_emotion INTEGER,               -- 实际情感强度
    is_wow_moment BOOLEAN DEFAULT FALSE,
    wow_type VARCHAR(30),
    position_order INTEGER,               -- 在全剧中的位置顺序
    created_at TIMESTAMP DEFAULT NOW()
);

-- 全局配置表（Layer 0锁定后的不变量）
CREATE TABLE project_config (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID REFERENCES projects(id) ON DELETE CASCADE,
    config_key VARCHAR(50) NOT NULL,      -- world_setting / magic_system / geography / history / culture / constraints / impossibilities
    config_value TEXT NOT NULL,
    is_locked BOOLEAN DEFAULT FALSE,
    locked_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(project_id, config_key)
);
```

```python
from typing import Optional

class ThinkingMode(Enum):
    NON_THINK = "non_think"

        self.deepseek_base = "https://api.deepseek.com/v1"
```

### 7.2 上下文组装器

```python
# backend/core/rag/retriever.py

class ContextBuilder:
    """素材Agent的核心：上下文组装器"""

    def __init__(self, db: AsyncSession):
        self.db = db
        self.MAX_CHARS = 30000

    async def build(self, project_id: str, scene_id: str) -> dict:
        """组装上下文包"""

        # P1: Layer 0 不变量（不裁剪）
        layer0 = await self._get_layer0(project_id)

        # P2: 前2个场景全文（不裁剪）
        prev_scenes = await self._get_prev_scenes(scene_id, count=2)

        # P3-P6: 按需包含，超出时裁剪
        layer1 = await self._get_layer1_state(project_id, scene_id)
        layer3 = await self._get_layer3_position(project_id, scene_id)
        layer4 = await self._get_layer4_chapter(project_id, scene_id)
        related_summaries = await self._get_related_summaries(project_id, scene_id)

        # 组装并检查总量
        sections = [
            {"name": "Layer 0不变量", "content": layer0, "priority": 1},
            {"name": "前序场景", "content": prev_scenes, "priority": 2},
            {"name": "角色状态", "content": layer1, "priority": 3},
            {"name": "章节上下文", "content": layer4, "priority": 4},
            {"name": "相关摘要", "content": related_summaries, "priority": 5},
            {"name": "主线脉络", "content": layer3, "priority": 6},
        ]

        total = sum(len(s["content"]) for s in sections)
        if total > self.MAX_CHARS:
            # 从低优先级开始裁剪
            sections.sort(key=lambda s: -s["priority"])
            for s in sections:
                if total <= self.MAX_CHARS:
                    break
                if s["priority"] >= 3:  # P1和P2不裁剪
                    excess = total - self.MAX_CHARS
                    cut = min(excess, len(s["content"]) - 200)
                    s["content"] = s["content"][:len(s["content"]) - cut]
                    total -= cut

        return {"sections": sections, "total_chars": total}
```

```python
# backend/tasks/scene_generation.py

from celery import shared_task

@shared_task(bind=True, max_retries=3, time_limit=600)
def generate_scene_task(self, project_id: str, scene_id: str, requirements: dict):
    """
    异步生成场景
    耗时操作，放到Celery执行，前端通过WebSocket或轮询获取进度
    """
    try:
        # 1. 组装上下文
        context = asyncio.run(ContextBuilder(db).build(project_id, scene_id))

        # 2. 选择思考模式
        mode = ThinkingMode.NON_THINK
        if requirements.get("is_wow_moment"):
            mode = ThinkingMode.THINK_MAX
        elif requirements.get("emotion_target", 5) >= 7:
            mode = ThinkingMode.THINK_HIGH

        # 3. 调用创作Agent
        result = asyncio.run(
            CreatorAgent(llm).generate_scene(context, requirements, mode)
        )

        # 4. 写入数据库
        asyncio.run(save_scene_draft(scene_id, result))

        return {"status": "completed", "scene_id": scene_id}

    except Exception as exc:
        self.retry(exc=exc, countdown=30)
```

```typescript
// frontend/src/hooks/useAgentProgress.ts

export function useAgentProgress(taskId: string) {
  const [progress, setProgress] = useState(null);

  useEffect(() => {
    const interval = setInterval(async () => {
      const res = await fetch(`/api/agents/tasks/${taskId}`);
      const data = await res.json();
      setProgress(data);
      if (data.status === 'completed' || data.status === 'failed') {
        clearInterval(interval);
      }
    }, 2000);  // 每2秒轮询

    return () => clearInterval(interval);
  }, [taskId]);

  return progress;
}
```

## 十二、前端需处理的边界情况

```

8. 生成失败: 显示具体错误+重试按钮
9. 审计死循环: 连续封驳>3次强制要求人类介入
10. 伏笔断裂: 实时在伏笔图谱上标红
```
